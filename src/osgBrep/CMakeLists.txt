
IF(DYNAMIC_OPENSCENEGRAPH)
    ADD_DEFINITIONS(-DOSGBREP_LIBRARY)

    ADD_DEFINITIONS(-DOSG_LIBRARY_STATIC)
ENDIF()

    ${HEADER_PATH}/Export

)

    Version.cpp

)

    osg
    OpenThreads
)
